create table Categoria (
   IdCategoria              number              not null,
   Nombre		        varchar2(20)        not null,
   constraint PK_CATEGORIA primary key (IdCategoria)
);

create table Ciudad (
   IdCiudad             number              not null,
   IdDepartamento       number              not null,
   Nombre               VARCHAR2(30)            not null,
   constraint PK_CIUDAD primary key (IdCiudad)
);

create table Departamento (
   IdDepartamento       number               not null,
   Nombre               VARCHAR2(30)             not null,
   constraint PK_DEPARTAMENTO primary key (IdDepartamento)
);

create table Envio (
   IdEnvio              number              not null,
   IdFactura            number              not null,
   constraint PK_ENVIO primary key (IdEnvio)
);

create table Factura (
   IdFactura            number              not null,
   IdUser               number              not null,
   IdSkin               number              not null,
   Fecha                date                not null,
   constraint PK_FACTURA primary key (IdFactura)
);

create table FacturaInsumos (
   IdFacturaInsumos     number              not null,
   IdMaterial           number              not null,
   IdProveedor          number              not null,
   Cantidad             number              not null,
   Costo                number(8,2)         not null,
   constraint PK_FACTURAINSUMOS primary key (IdFacturaInsumos)
);

create table ImagenUser (
   IdImagenUser         number              not null,
   URLImagen            number(1)            not null,
   constraint PK_IMAGENUSER primary key (IdImagenUser)
);

create table Marca (
   IdMarca              number              not null,
   IdModelo             number              not null,
   Nombre               varchar2(30)        not null,
   constraint PK_MARCA primary key (IdMarca)
);

create table Material (
   IdMaterial           number              not null,
   Nombre		varchar2(20)		not null,
   Descripcion          varchar2(50)            not null,
   constraint PK_MATERIAL primary key (IdMaterial)
);

create table Modelo (
   IdModelo             number               not null,
   Modelo		varchar2(30)	     not null,
   Dimensiones          varchar2(30)             not null,
   Disponible           number(1)            not null,
   CostoSkin            number(8,2)          not null,
   Imagen               number(1)            not null,
   constraint PK_MODELO primary key (IdModelo)
);

create table Proveedor (
   IdProveedor          number               not null,
   NIT                  varchar2(15)             not null,
   Direccion            varchar2(30)             not null,
   Correo               varchar2(30)             not null,
   Telefono             varchar2(30)             not null,
   constraint PK_PROVEEDOR primary key (IdProveedor)
);

create table Skin (
   IdSkin               number              not null,
   IdCategoria          number              not null,
   IdImagenUuser        number              not null,
   IdUser               number              not null,
   Existencia           number(1)           not null,
   Imagen               number(1)                not null,
   constraint PK_SKIN primary key (IDSKIN)
);

create table SolicitudSkin (
   IdSolicitud          number              not null,
   IdMarca              number              not null,
   IdTipo               number              not null,
   IdUser               number              not null,
   USE_IDUSER           number              not null,
   IdSkin               number              not null,
   CostoSkin            number(8,2)         not null,
   constraint PK_SOLICITUDSKIN primary key (IdSolicitud)
);

create table Tipo (
   IdTipo               number              not null,
   Nombre               varchar2(30)          not null,
   constraint PK_TIPO primary key (IdTipo)
);

create table Usuario (
   IdUser               number              not null,
   IdCiudad             number              not null,
   Tipo                 varchar(1)           not null,
   Nombres              varchar2(30)            not null,
   Apellidos            varchar2(30)            not null,
   Documento            varchar2(30)            not null,
   Telefono             varchar2(30)            not null,
   Direccion            varchar2(50)            not null,
   Email                varchar2(50)            not null,
   Contrasena           varchar2(30)            not null,
   Fecha                date                null,
   Estado               number(1)           null,
   constraint PK_USER primary key (IdUser)
);
